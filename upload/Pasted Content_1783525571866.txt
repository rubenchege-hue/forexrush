function calcPnL(p) {
  const d=pd[p.pair]; if(!d) return 0;
  const c=P[p.pair], m=d.cp, dir=p.side==='Long'?1:-1;
  if(c.ut==='coins') return (m-p.entry)*p.size*dir;
  let pnl=(m-p.entry)*p.size*c.ls*dir;
  if(c.q==='JPY') pnl/=m; // JPY profit → USD
  return pnl;
}

function calcMargin(p) {
  const c=P[p.pair];
  return c.ut==='coins' ? (p.size*p.entry)/p.lev : (p.size*c.ls*p.entry)/p.lev;
}

function totalPnL() { return pos.reduce((s,p)=>s+calcPnL(p),0); }
function totalMargin() { return pos.reduce((s,p)=>s+calcMargin(p),0); }

function genCandles(n,start,tf,vol) {
  const d=[]; let pr=start; const now=Date.now();
  for(let i=0;i<n;i++){
    const dr=(Math.random()-.48)*vol, o=pr, c=o+dr;
    const h=Math.max(o,c)+Math.random()*vol*.8, l=Math.min(o,c)-Math.random()*vol*.8;
    d.push({o,h,l,c,v:50+Math.random()*450,t:now-(n-i)*tf*60000});
    pr=c;
  }
  return d;
}

function genOB(pr,sp,dec) {
  const bids=[],asks=[],hs=sp/2;
  for(let i=0;i<14;i++){
    const bp=pr-hs-(i+1)*(sp*(.5+Math.random()*2));
    const ap=pr+hs+(i+1)*(sp*(.5+Math.random()*2));
    const ba=+(Math.random()*1.5+.05).toFixed(dec);
    const aa=+(Math.random()*1.5+.05).toFixed(dec);
    bids.push({p:+bp.toFixed(dec),a:ba}); asks.push({p:+ap.toFixed(dec),a:aa});
  }
  let bt=0; bids.forEach(b=>{bt+=b.a;b.t=+bt.toFixed(dec);});
  let at=0; asks.forEach(a=>{at+=a.a;a.t=+at.toFixed(dec);});
  const mx=Math.max(bids[bids.length-1]?.t||.01,asks[asks.length-1]?.t||.01,.01);
  return {bids,asks,sp:+sp.toFixed(dec),mx};
}

  ctx.beginPath();ctx.moveTo(0,lY);ctx.lineTo(aW,lY);ctx.stroke();ctx.setLineDash([]);
  ctx.fillStyle=lg?'#00e87b':'#ff3b5c';rR(ctx,aW,lY-9,paW,18,3);ctx.fill();
  ctx.fillStyle='#0a0a0f';ctx.font='bold 9px "JetBrains Mono"';ctx.textAlign='center';
  ctx.fillText(fp(d.cp),aW+paW/2,lY+3);

  // Time axis
  ctx.fillStyle='#404060';ctx.font='9px "JetBrains Mono"';ctx.textAlign='center';
  const tS=Math.max(1,Math.floor(vC/8));
  for(let i=0;i<vC;i+=tS){const x=iX(i),dt=new Date(vis[i].t);ctx.fillText(`${dt.getHours().toString().padStart(2,'0')}:${dt.getMinutes().toString().padStart(2,'0')}`,x,aH+14);ctx.strokeStyle='rgba(40,40,64,.3)';ctx.lineWidth=.5;ctx.beginPath();ctx.moveTo(x,0);ctx.lineTo(x,cH);ctx.stroke();}

  // Crosshair
  if(xh.x!==null&&xh.y!==null&&xh.x<aW&&xh.y<cH){
    ctx.setLineDash([3,3]);ctx.strokeStyle='rgba(200,245,66,.3)';ctx.lineWidth=.5;
    ctx.beginPath();ctx.moveTo(xh.x,0);ctx.lineTo(xh.x,cH);ctx.stroke();
    ctx.beginPath();ctx.moveTo(0,xh.y);ctx.lineTo(aW,xh.y);ctx.stroke();ctx.setLineDash([]);
    const hp=mn+(1-xh.y/cH)*(mx-mn);
    ctx.fillStyle='#c8f542';rR(ctx,aW,xh.y-9,paW,18,3);ctx.fill();
    ctx.fillStyle='#0a0a0f';ctx.font='bold 9px "JetBrains Mono"';ctx.textAlign='center';
    ctx.fillText(fp(hp),aW+paW/2,xh.y+3);
    const ci=Math.floor(xh.x/gap);
    if(ci>=0&&ci<vC){
      const dt=new Date(vis[ci].t),tl=`${dt.getHours().toString().padStart(2,'0')}:${dt.getMinutes().toString().padStart(2,'0')}:${dt.getSeconds().toString().padStart(2,'0')}`;
      ctx.fillStyle='rgba(200,245,66,.8)';const tw=cx.measureText(tl).width+10;
      rR(ctx,xh.x-tw/2,aH,tw,18,3);ctx.fill();ctx.fillStyle='#0a0a0f';ctx.fillText(tl,xh.x,aH+13);
      const c=vis[ci];
      document.getElementById('oO').textContent=fp(c.o);
      document.getElementById('oH').textContent=fp(c.h);
      document.getElementById('oL').textContent=fp(c.l);
      document.getElementById('oC').textContent=fp(c.c);
      document.getElementById('oC').style.color=c.c>=c.o?'var(--grn)':'var(--red)';
      document.getElementById('oV').textContent=c.v.toFixed(1);
    }
  }
  ctx.strokeStyle='#282840';ctx.lineWidth=1;ctx.strokeRect(.5,.5,aW-.5,aH-.5);
}
